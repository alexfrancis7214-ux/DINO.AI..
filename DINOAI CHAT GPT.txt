function handleChat(text){
  const t = text.toLowerCase();
  if(t.includes("joke")) return addChat("bot","Why did the dino cross the road? Because the chicken hadn‚Äôt been invented yet!");
  if(t.includes("avatar")) return addChat("bot","Try prompts like: 'green dino with sunglasses' or 'small blue dino with hat'.");
  if(t.includes("hello")||t.includes("hi")) return addChat("bot","Hi! I'm Dino ü¶ñ ‚Äî I can draw, play music, and run games.");
  if(t.includes("play") && t.includes("music")) return addChat("bot","Open the Music tab and type 'play' to hear a melody.");
  // fallback: friendly echo with tiny "AI" flair
  const replies = [
    "Roar! That sounds interesting.",
    "I like that ‚Äî tell me more!",
    "Rawr! Let's try that in Avatar or Music.",
    "Dino thinks that's cool!"
  ];
  addChat("bot", replies[Math.floor(Math.random()*replies.length)]);
}

const avatarCanvas = document.getElementById("avatarCanvas");
const actx = avatarCanvas.getContext("2d");
function drawDino(opts){
  const w = avatarCanvas.width, h = avatarCanvas.height;
  actx.clearRect(0,0,w,h);
  // background circle
  actx.fillStyle = "rgba(255,255,255,0.02)";
  actx.beginPath(); actx.arc(w/2,h/2,150,0,Math.PI*2); actx.fill();

  // body
  actx.fillStyle = opts.color;
  actx.beginPath();
  actx.ellipse(w/2, h*0.6, opts.bodyW, opts.bodyH, 0, 0, Math.PI*2);
  actx.fill();

  // head
  actx.beginPath();
  actx.arc(w/2, h*0.36, opts.headR, 0, Math.PI*2);
  actx.fill();

  // belly
  actx.fillStyle = lighten(opts.color,0.25);
  actx.beginPath();
  actx.ellipse(w/2, h*0.66, opts.bodyW*0.6, opts.bodyH*0.5, 0, 0, Math.PI*2);
  actx.fill();

  // eye
  actx.fillStyle = "#fff";
  actx.beginPath(); actx.arc(w/2 - opts.headR*0.3, h*0.34, opts.eyeR, 0, Math.PI*2); actx.fill();
  actx.beginPath(); actx.arc(w/2 + opts.headR*0.3, h*0.34, opts.eyeR, 0, Math.PI*2); actx.fill();
  actx.fillStyle = "#000";
  actx.beginPath(); actx.arc(w/2 - opts.headR*0.3, h*0.34, opts.pupilR, 0, Math.PI*2); actx.fill();
  actx.beginPath(); actx.arc(w/2 + opts.headR*0.3, h*0.34, opts.pupilR, 0, Math.PI*2); actx.fill();

  // spikes
  if(opts.spikes){
    const spikeCount = 6;
    for(let i=0;i<spikeCount;i++){
      const x = w/2 - opts.bodyW*0.8 + (i*(opts.bodyW*1.6/(spikeCount-1)));
      const y = h*0.45 - Math.abs(i - spikeCount/2)*4;
      actx.fillStyle = darken(opts.color,0.15);
      actx.beginPath();
      actx.moveTo(x,y);
      actx.lineTo(x+12,y-24);
      actx.lineTo(x+24,y);
      actx.closePath();
      actx.fill();
    }
  }

  // sunglasses
  if(opts.sunglasses){
    actx.fillStyle = "#000";
    actx.fillRect(w/2 - 36, h*0.32 - 10, 28, 20);
    actx.fillRect(w/2 + 8, h*0.32 - 10, 28, 20);
    actx.fillStyle = "#222";
    actx.fillRect(w/2 - 8, h*0.32 - 3, 16, 6);
  }

  // hat
  if(opts.hat){
    actx.fillStyle = opts.hatColor || "#7a3e1d";
    actx.fillRect(w/2 - 56, h*0.36 - opts.headR - 18, 112, 18);
    actx.fillStyle = darken(opts.hatColor || "#7a3e1d",0.12);
    actx.fillRect(w/2 - 30, h*0.36 - opts.headR - 40, 60, 24,);
  }

  // smile
  actx.strokeStyle = "#000";
  actx.lineWidth = 2;
  actx.beginPath();
  actx.arc(w/2, h*0.42, 22, 0.1*Math.PI, 0.9*Math.PI);
  actx.stroke();
}
function lighten(col, amt){ // simple hex->lighter
  try{
    const c = hexToRgb(col);
    return `rgb(${Math.min(255,Math.round(c.r + (255-c.r)*amt))},${Math.min(255,Math.round(c.g + (255-c.g)*amt))},${Math.min(255,Math.round(c.b + (255-c.b)*amt))})`;
  }catch(e){ return col; }
}
function darken(col, amt){
  try{
    const c = hexToRgb(col);
    return `rgb(${Math.max(0,Math.round(c.r*(1-amt)))},${Math.max(0,Math.round(c.g*(1-amt)))},${Math.max(0,Math.round(c.b*(1-amt)))})`;
  }catch(e){ return col; }
}
function hexToRgb(hex){
  hex = hex.replace("#","").trim();
  if(hex.length===3) hex = hex.split("").map(ch=>ch+ch).join("");
  const int = parseInt(hex,16);
  return { r:(int>>16)&255, g:(int>>8)&255, b:int&255 };
}
function parseAvatarPrompt(prompt){
  const p = prompt.toLowerCase();
  // color
  let color = "#40e6a8";
  if(p.includes("red")) color = "#ff6b6b";
  if(p.includes("green")) color = "#40e6a8";
  if(p.includes("blue")) color = "#5bb8ff";
  if(p.includes("yellow")) color = "#ffd166";
  if(p.includes("gold")) color = "#f4c150";
  if(p.includes("purple")) color = "#c389ff";
  // size
  let small = p.includes("small");
  let large = p.includes("large");
  const bodyW = large ? 150 : small ? 70 : 100;
  const bodyH = large ? 80 : small ? 40 : 60;
  const headR = large ? 70 : small ? 30 : 50;
  const eyeR = Math.max(4, Math.round(headR*0.15));
  const pupilR = Math.max(2, Math.round(headR*0.07));
  return {
    color, bodyW, bodyH, headR, eyeR, pupilR,
    sunglasses: p.includes("sunglass")||p.includes("glasses"),
    hat: p.includes("hat"),
    hatColor: p.includes("red hat") ? "#d34d4d" : p.includes("blue hat") ? "#3b86ff" : "#7a3e1d",
    spikes: p.includes("spike")||p.includes("spikes")||p.includes("spiny")
  };
}

  const opts = parseAvatarPrompt(prompt);
  drawDino(opts);
});

let audioCtx = null;
let masterGain = null;
let currentNodes = [];
let tempo = 1;
let instrument = 'sine';
const nowPlaying = document.getElementById("nowPlaying");
document.getElementById("musicSend").addEventListener("click", ()=>{
  const cmd = document.getElementById("musicInput").value.trim();
  if(!cmd) return;
  handleMusicCommand(cmd);
  document.getElementById("musicInput").value = "";
});
function handleMusicCommand(cmd){
  const c = cmd.toLowerCase();
  if(c.startsWith("play")){
    if(!audioCtx) audioCtx = new (window.AudioContext||window.webkitAudioContext)();
    if(!masterGain){ masterGain = audioCtx.createGain(); masterGain.gain.value = 0.12; masterGain.connect(audioCtx.destination); }
    stopMusic();
    playMelody();
    nowPlaying.textContent = "Playing melody ‚Äî instrument: "+instrument+" ‚Äî tempo: "+tempo;
  } else if(c.startsWith("stop")){
    stopMusic();
    nowPlaying.textContent = "Stopped.";
  } else if(c.startsWith("tempo")){
    const parts = c.split(/\s+/);
    const v = parseFloat(parts[1]);
    if(!isNaN(v) && v>0){ tempo = v; nowPlaying.textContent = "Tempo set to "+tempo; }
  } else if(c.startsWith("instrument")){
    const parts = c.split(/\s+/);
    if(parts[1]){ instrument = parts[1]; nowPlaying.textContent = "Instrument set to "+instrument; }
  } else {
    nowPlaying.textContent = "Unknown command.";
  }
}
function playMelody(){
  // simple melody pattern
  const notes = [ 261.63, 293.66, 329.63, 349.23, 392.00, 440.00, 493.88, 523.25 ];
  let t = audioCtx.currentTime + 0.05;
  const beat = 0.35 / tempo;
  notes.forEach((freq,i)=>{
    const osc = audioCtx.createOscillator();
    osc.type = instrument;
    osc.frequency.value = freq;
    const g = audioCtx.createGain();
    g.gain.setValueAtTime(0, t + i*beat);
    g.gain.linearRampToValueAtTime(1*0.8, t + i*beat + 0.02);
    g.gain.exponentialRampToValueAtTime(0.01, t + (i+1)*beat);
    osc.connect(g); g.connect(masterGain);
    osc.start(t + i*beat);
    osc.stop(t + (i+1)*beat + 0.02);
    currentNodes.push(osc);
  });
  // schedule clean-up
  setTimeout(()=>{ currentNodes = []; }, (notes.length+1)*beat*1000);
}
function stopMusic(){
  if(currentNodes.length>0){
    currentNodes.forEach(n=>{ try{ n.stop(); }catch(e){} });
    currentNodes = [];
  }
}

const gameCanvas = document.getElementById("gameCanvas");
const gctx = gameCanvas.getContext("2d");
let gw = gameCanvas.width, gh = gameCanvas.height;
function resizeGame(){ gw = gameCanvas.width = gameCanvas.clientWidth * devicePixelRatio; gh = gameCanvas.height = gameCanvas.clientHeight * devicePixelRatio; scale = devicePixelRatio; }
window.addEventListener('resize', ()=>{ resizeGame(); drawStatic(); });
resizeGame();

let gameState = "stopped"; // stopped|running|over
let player = {x:80, y:0, vy:0, w:36, h:36, grounded:false};
let gravity = 1500;
let groundY = 140 * devicePixelRatio;
let obstacles = [];
let spawnTimer = 0;
let score = 0;
let lastTime = null;
let scale = devicePixelRatio;

function resetGame(){
  obstacles = [];
  player.y = groundY - player.h;
  player.vy = 0;
  player.grounded = true;
  spawnTimer = 0; score = 0; lastTime = null;
  gameState = "stopped";
  document.getElementById("gameScore").innerText = "Score: 0";
  drawStatic();
}
function startGame(){
  if(gameState === "running") return;
  resetGame();
  gameState = "running";
  lastTime = performance.now();
  requestAnimationFrame(gameLoop);
}
document.getElementById("startGame").addEventListener("click", startGame);
document.getElementById("resetGame").addEventListener("click", resetGame);

function spawnObstacle(){
  const h = (20 + Math.random()*40) * scale;
  const w = (12 + Math.random()*20) * scale;
  obstacles.push({x: gw + 20, y: groundY - h, w, h});
}
function gameLoop(ts){
  if(gameState !== "running") return;
  const dt = Math.min(100, ts - (lastTime || ts)) / 1000; lastTime = ts;
  // physics
  player.vy += gravity * dt;
  player.y += player.vy * dt;
  if(player.y >= groundY - player.h){ player.y = groundY - player.h; player.vy = 0; player.grounded = true; }
  // spawn obstacles
  spawnTimer -= dt;
  if(spawnTimer <= 0){ spawnObstacle(); spawnTimer = 0.9 - Math.min(0.6, score/200); }
  // move obstacles
  const speed = 220 + Math.min(400, score*0.6);
  for(let i=obstacles.length-1;i>=0;i--){
    obstacles[i].x -= speed * dt;
    if(obstacles[i].x + obstacles[i].w < -50) obstacles.splice(i,1);
    // collision
    if(rectOverlap(player.x, player.y, player.w, player.h, obstacles[i].x, obstacles[i].y, obstacles[i].w, obstacles[i].h)){
      gameOver(); return;
    }
  }
  // scoring
  score += dt * 100;
  document.getElementById("gameScore").innerText = "Score: " + Math.floor(score);
  // draw
  drawScene();
  requestAnimationFrame(gameLoop);
}
function rectOverlap(x1,y1,w1,h1,x2,y2,w2,h2){
  return !(x1+w1 < x2 || x1 > x2+w2 || y1+h1 < y2 || y1 > y2+h2);
}
function gameOver(){
  gameState = "over";
  // simple flash
  gctx.fillStyle = "rgba(255,0,0,0.12)";
  gctx.fillRect(0,0,gw,gh);
  document.getElementById("gameScore").innerText += "  ‚Äî Game Over";
}
function drawStatic(){
  // draw static scene when idle
  gctx.clearRect(0,0,gw,gh);
  // sky
  gctx.fillStyle = "#9ad0ff";
  gctx.fillRect(0,0,gw,gh);
  // ground
  gctx.fillStyle = "#6b4b2b";
  gctx.fillRect(0, groundY, gw, gh-groundY);
  // player
  drawPlayer();
}
function drawScene(){
  gctx.clearRect(0,0,gw,gh);
  // sky
  gctx.fillStyle = "#9ad0ff";
  gctx.fillRect(0,0,gw,gh);
  // ground
  gctx.fillStyle = "#6b4b2b";
  gctx.fillRect(0, groundY, gw, gh-groundY);
  // obstacles
  gctx.fillStyle = "#227722";
  obstacles.forEach(o=>{
    gctx.fillRect(o.x, o.y, o.w, o.h);
  });
  // player
  drawPlayer();
  // score
  gctx.fillStyle = "#fff";
  gctx.font = `${16*scale}px Arial`;
  gctx.fillText("Score: " + Math.floor(score), 10*scale, 24*scale);
}
function drawPlayer(){
  gctx.save();
  gctx.translate(player.x, player.y);
  // body
  gctx.fillStyle = "#40e6a8";
  gctx.fillRect(0,0,player.w,player.h);
  // eye
  gctx.fillStyle = "#fff";
  gctx.fillRect(player.w*0.55, player.h*0.2, player.w*0.2, player.h*0.2);
  gctx.fillStyle = "#000";
  gctx.fillRect(player.w*0.6, player.h*0.25, player.w*0.07, player.h*0.07);
  gctx.restore();
}

function jump(){

const storiesGrid = document.getElementById("storiesGrid");
function renderStories(list){
  storiesGrid.innerHTML = "";
  list.forEach(s=>{
    const d = document.createElement("div");
    d.className = "story-card";
    d.textContent = s;
    storiesGrid.appendChild(d);
  });
}
renderStories(stories);

document.getElementById("storyReload").addEventListener("click", ()=>{
  const shuffled = stories.slice().sort(()=>Math.random()-0.5);
  renderStories(shuffled.slice(0,200));
});
document.getElementById("storySort").addEventListener("change", (e)=>{
  const v = e.target.value;
  let out = stories.slice();
  if(v==="az") out.sort();
  else if(v==="za") out.sort().reverse();
  else if(v==="len") out.sort((a,b)=>a.length-b.length);
  renderStories(out);
});
document.getElementById("storySearch").addEventListener("input",(e)=>{
  const q = e.target.value.toLowerCase().trim();
  if(!q) return renderStories(stories);
  const filtered = stories.filter(s=>s.toLowerCase().includes(q));
  renderStories(filtered);
});
